        // Video Modal Functionality
        const videoItems = document.querySelectorAll('.video-item');
        const modal = document.getElementById('videoModal');
        const modalVideo = document.getElementById('modalVideo');
        const modalVideoSource = document.getElementById('modalVideoSource');
        const modalTitle = document.getElementById('modalVideoTitle');
        const modalDescription = document.getElementById('modalVideoDescription');
        const closeBtn = document.getElementById('modalClose');
        const prevBtn = document.getElementById('prevVideo');
        const nextBtn = document.getElementById('nextVideo');
        
        let currentVideoIndex = 0;
        
        const videoData = [
            {
                src: 'assets/How-to-keep-your-flow-and-focus-with-Microsoft-Edge.mp4',
                title: 'Flow & Focus',
                description: 'Demonstrating productivity features that help users maintain concentration and workflow in their daily browsing.'
            },
            {
                src: 'assets/How-to-stay-organized-with-Microsoft-Edge.mp4',
                title: 'Stay Organized',
                description: 'Showcasing organizational tools and features for better browsing management and digital productivity.'
            },
            {
                src: 'assets/How-to-stay-safer-online-with-Microsoft-Edge.mp4',
                title: 'Online Safety',
                description: 'Highlighting security features and safe browsing practices that protect users in the digital world.'
            },
            {
                src: 'assets/How-to-use-Bing-AI-in-Microsoft-Edge-to-get-inspired-and-boost-creativity.mp4',
                title: 'AI-Powered Creativity',
                description: 'Introducing AI-powered features for creative inspiration and intelligent assistance in everyday tasks.'
            },
            {
                src: 'assets/How-to-save_money.mp4',
                title: 'Save Money',
                description: 'Discover how Edge\'s smart shopping features help you find deals, compare prices, and save money while browsing.'
            },
            {
                src: 'assets/how-to-save_time_shopping-online.mp4',
                title: 'Save Time Shopping',
                description: 'Learn time-saving shopping techniques and features that streamline your online purchasing experience.'
            }
        ];
        
        // Open modal
        videoItems.forEach((item, index) => {
            item.addEventListener('click', function(e) {
                e.preventDefault();
                currentVideoIndex = index;
                openModal();
            });
        });
        
        function openModal() {
            const video = videoData[currentVideoIndex];
            modalVideoSource.src = video.src;
            modalVideo.load();
            modalTitle.textContent = video.title;
            modalDescription.textContent = video.description;
            modal.classList.add('active');
            document.body.style.overflow = 'hidden';
        }
        
        function closeModal() {
            modal.classList.remove('active');
            modalVideo.pause();
            document.body.style.overflow = 'auto';
        }
        
        function previousVideo() {
            currentVideoIndex = (currentVideoIndex - 1 + videoData.length) % videoData.length;
            openModal();
        }
        
        function nextVideo() {
            currentVideoIndex = (currentVideoIndex + 1) % videoData.length;
            openModal();
        }
        
        // Event listeners
        closeBtn.addEventListener('click', closeModal);
        prevBtn.addEventListener('click', previousVideo);
        nextBtn.addEventListener('click', nextVideo);
        
        // Close modal when clicking outside
        modal.addEventListener('click', function(e) {
            if (e.target === modal) {
                closeModal();
            }
        });
        
        // Keyboard navigation
        document.addEventListener('keydown', function(e) {
            if (modal.classList.contains('active')) {
                if (e.key === 'Escape') {
                    closeModal();
                } else if (e.key === 'ArrowLeft') {
                    previousVideo();
                } else if (e.key === 'ArrowRight') {
                    nextVideo();
                }
            }
        });
        
        // Scroll functionality
        const scrollPrompt = document.getElementById('scroll-prompt');
        const backToTop = document.getElementById('back-to-top');
        
        // Show/hide scroll prompt and back to top button
        window.addEventListener('scroll', function() {
            const scrollY = window.scrollY;
            
            if (scrollY > 100) {
                scrollPrompt.style.opacity = '0';
                scrollPrompt.style.pointerEvents = 'none';
            } else {
                scrollPrompt.style.opacity = '1';
                scrollPrompt.style.pointerEvents = 'auto';
            }
            
            if (scrollY > window.innerHeight / 2) {
                backToTop.style.opacity = '1';
                backToTop.style.pointerEvents = 'auto';
            } else {
                backToTop.style.opacity = '0';
                backToTop.style.pointerEvents = 'none';
            }
        });
        
        // Scroll prompt click
        scrollPrompt.addEventListener('click', function() {
            document.querySelector('.py-24').scrollIntoView({ 
                behavior: 'smooth' 
            });
        });
        
        // Back to top click
        backToTop.addEventListener('click', function() {
            window.scrollTo({ 
                top: 0, 
                behavior: 'smooth' 
            });
        });
        
        // Mobile menu functionality
        const mobileMenuBtn = document.getElementById('mobile-menu-btn');
        const mobileMenu = document.createElement('div');
        mobileMenu.className = 'fixed top-20 left-0 right-0 bg-gray-900 bg-opacity-95 backdrop-filter backdrop-blur-lg border-b border-gray-700 p-6 space-y-4 z-40 hidden';
        mobileMenu.innerHTML = `
            <a href="#" class="block text-gray-300 hover:text-red-400 transition-colors font-medium py-2">Home</a>

            <a href="#" class="block text-gray-300 hover:text-red-400 transition-colors font-medium py-2">Contact</a>
        `;
        
        document.body.appendChild(mobileMenu);
        
        mobileMenuBtn.addEventListener('click', function() {
            mobileMenu.classList.toggle('hidden');
        });
        
        // Smooth scroll for all anchor links
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();
                const target = document.querySelector(this.getAttribute('href'));
                if (target) {
                    target.scrollIntoView({
                        behavior: 'smooth'
                    });
                }
            });
        });
